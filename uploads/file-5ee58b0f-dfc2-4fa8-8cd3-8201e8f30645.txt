Odd or Even
#!/bin/bash

echo "Enter a number:"
read number

if (( number % 2 == 0 )); then
    echo "$number is even."
else
    echo "$number is odd."
fi



Calculator
#!/bin/bash 

echo "Enter two numbers:"
read num1
read num2

sum=$((num1 + num2))
echo "Sum: $sum"

difference=$((num1 - num2))
echo "Difference: $difference"

product=$((num1 * num2))
echo "Product: $product"

if (( num2 != 0 )); then
    quotient=$((num1 / num2))
    echo "Quotient: $quotient"
else
    echo "Division by zero error."
fi



Factorial
#!/bin/bash

echo "Enter a number:"
read num

factorial=1
for ((i = 1; i <= num; i++)); do
    factorial=$((factorial * i))
done

echo "Factorial of $num is $factorial"



Sum of n numbers:
#!/bin/bash

echo "Enter the number of elements:"
read n

sum=0
echo "Enter $n numbers:"
for ((i = 0; i < n; i++)); do
    read num
    sum=$((sum + num))
done

echo "Sum of $n numbers is $sum"



Greatest of 3 numbers:
#!/bin/bash

echo "Enter three numbers:"
read num1
read num2
read num3

if ((num1 >= num2 && num1 >= num3)); then
    echo "$num1 is the greatest."
elif ((num2 >= num1 && num2 >= num3)); then
    echo "$num2 is the greatest."
else
    echo "$num3 is the greatest."
fi
